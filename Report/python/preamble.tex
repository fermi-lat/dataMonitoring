
% Document class.
\documentclass[oneside, 12pt, letterpaper]{report}

% Required packages.
\usepackage{graphicx}
\usepackage{rotating}
\usepackage{color}
\usepackage[colorlinks=true]{hyperref}

% Document geometry.
\textwidth = 19.3 truecm
\textheight = 25.6 truecm
\oddsidemargin -1.45 truecm
\topmargin -2.75 truecm

% Vertical label to be put on the left/right of the plots.
% Usage:
% \gplotlabel{half-height}{text}
\newcommand{\gplotleftlabel}[2]{%
\raisebox{#1}{\parbox[c]{35pt}{\fontsize{4}{8} \selectfont #2}}
}
\newcommand{\gplotrightlabel}[2]{%
\raisebox{#1}{\parbox[c]{15pt}{\fontsize{4}{8} \selectfont #2}}
}

% Vertical label to be put on the left of each panel.
% Usage:
% \gpanellabel{width}{text}
\newcommand{\gpanellabel}[2]{%
\begin{minipage}[c]{#1}\begin{sideways}\fontsize{7}{15} \selectfont #2%
\end{sideways}\end{minipage}%
}

% Panel plot.
% Usage:
% \gpanelplot{width}{topmargin}{content}
\newcommand{\gpanelplot}[3]{%
$\left\{ \begin{minipage}[c]{#1}\rule{0 pt}{#2} #3\end{minipage}\right.$%
}
