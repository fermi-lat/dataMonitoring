% Vertical label to be put on the left/right of the plots.
% Usage:
% \gplotlabel{half-height}{text}
\newcommand{\gplotleftlabel}[2]{%
\raisebox{#1}{\parbox[c]{35pt}{\fontsize{4}{6} \selectfont #2}}
}
\newcommand{\gplotrightlabel}[2]{%
\raisebox{#1}{\parbox[c]{70pt}{\fontsize{4}{6} \selectfont #2}}
}

% Panel plot.
% Usage:
% \gpanelplot{width}{topmargin}{content}
\newcommand{\gpanelplot}[3]{%
\hspace*{-15pt}
\begin{minipage}[c]{#1}\rule{0pt}{#2} #3\end{minipage}%
}
